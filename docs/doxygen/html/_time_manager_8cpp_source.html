112bool TimeManager::setTimeFromNTP(const char* ntpServer) {
113 if (ntpServer) {
114 _ntpServer = String(ntpServer);
115 _initializeNTP();

123 if (_timeClient->update()) {

125 DateTime ntpTime = DateTime(epochTime);

143bool TimeManager::syncSystemTimeFromRTC() {
144 if (!_rtcConnected) {
145 Serial.println("TimeManager: Cannot sync system time - RTC not connected");
146 return false;
147 }
148
149 DateTime rtcTime = _rtc.now();
150
151 // Convert RTC time to timeval struct for settimeofday
152 struct timeval tv;
153 tv.tv_sec = rtcTime.unixtime();
154 tv.tv_usec = 0;
155
156 // Set the system time
157 if (settimeofday(&tv, NULL) == 0) {
158 Serial.printf("TimeManager: System time synced from RTC: %04d-%02d-%02d %02d:%02d:%02d\n",
159 rtcTime.year(), rtcTime.month(), rtcTime.day(),
160 rtcTime.hour(), rtcTime.minute(), rtcTime.second());
161 return true;
162 } else {
163 Serial.println("TimeManager: Failed to set system time");
164 return false;
165 }
166}

273void TimeManager::update() {
274 // Auto-sync with NTP if enabled and interval elapsed
275 if (_ntpSyncEnabled && _isWiFiConnected() &&
276 (millis() - _lastNTPSync) > _ntpUpdateInterval &&
277 (millis() - _lastNTPAttempt) > 60000) { // Don't attempt more than once per minute
278
279 _lastNTPAttempt = millis();
280 if (setTimeFromNTP()) {
281 Serial.println("TimeManager: Automatic NTP sync successful");
282 }
283 }

432void TimeManager::loadConfig() {
433 File file = LittleFS.open("/time_config.json", "r");
434 if (file) {
435 DynamicJsonDocument doc(256);
436 DeserializationError error = deserializeJson(doc, file);
437 file.close();
438
439 if (!error) {
440 _timezoneOffset = doc["timezone_offset"] | 0;
441 _ntpServer = doc["ntp_server"] | "pool.ntp.org";
442 _ntpUpdateInterval = doc["ntp_update_interval"] | 3600000;
443 _ntpSyncEnabled = doc["ntp_sync_enabled"] | true;
444
445 // Reinitialize NTP with loaded settings
446 _initializeNTP();

466String TimeManager::_formatDateTime(DateTime dt, const String& format) {
467 String result = format;
468
469 // Replace format tokens
470 result.replace("YYYY", String(dt.year()));
471 result.replace("MM", String(dt.month()).length() == 1 ? "0" + String(dt.month()) : String(dt.month()));
472 result.replace("DD", String(dt.day()).length() == 1 ? "0" + String(dt.day()) : String(dt.day()));
473 result.replace("hh", String(dt.hour()).length() == 1 ? "0" + String(dt.hour()) : String(dt.hour()));
474 result.replace("mm", String(dt.minute()).length() == 1 ? "0" + String(dt.minute()) : String(dt.minute()));
475 result.replace("ss", String(dt.second()).length() == 1 ? "0" + String(dt.second()) : String(dt.second()));
476
477 return result;
478}
