225 // Auto-sync with NTP if enabled and interval elapsed
226 if (_ntpSyncEnabled && _isWiFiConnected() &&
227 (millis() - _lastNTPSync) > _ntpUpdateInterval &&
228 (millis() - _lastNTPAttempt) > 60000) { // Don't attempt more than once per minute
