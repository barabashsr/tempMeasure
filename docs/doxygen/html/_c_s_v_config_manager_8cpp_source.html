57void CSVConfigManager::_exportPointToCSV(String& csv, MeasurementPoint* point, const String& pointType) {
58 if (!point) return;
59
60 // Get bound sensor info
61 Sensor* sensor = point->getBoundSensor();
62 String romString = "";
63 String busNumber = "";
64
65 if (sensor) {
66 if (sensor->getType() == SensorType::DS18B20) {
67 romString = sensor->getDS18B20RomString();
68 } else if (sensor->getType() == SensorType::PT1000) {
69 busNumber = String(_controller.getSensorBus(sensor)); // Changed from ChipSelectPin to BusNumber
70 }
71 }
72
73 // Build the CSV row
74 csv += String(point->getAddress()) + ",";
75 csv += _escapeCSVField(point->getName()) + ",";
76 csv += pointType + ",";
77 csv += String(point->getCurrentTemp()) + ",";
78 csv += String(point->getMinTemp()) + ",";
79 csv += String(point->getMaxTemp()) + ",";
80 csv += String(point->getLowAlarmThreshold()) + ",";
81 csv += String(point->getHighAlarmThreshold()) + ",";
82 csv += _escapeCSVField(romString) + ",";
83 csv += busNumber + ","; // Changed variable name
84
85 // Add alarm priorities for each type
86 csv += _getAlarmPriorityForPoint(point->getAddress(), AlarmType::HIGH_TEMPERATURE) + ",";
87 csv += _getAlarmPriorityForPoint(point->getAddress(), AlarmType::LOW_TEMPERATURE) + ",";
88 csv += _getAlarmPriorityForPoint(point->getAddress(), AlarmType::SENSOR_ERROR) + ",";
89 csv += _getAlarmPriorityForPoint(point->getAddress(), AlarmType::SENSOR_DISCONNECTED) + "\n";
90}

93bool CSVConfigManager::importPointsWithAlarmsFromCSV(const String& csvData) {
94 if (!validatePointsCSV(csvData)) {
95 return false;
96 }
97
98 int lineStart = 0;
99 int lineEnd = csvData.indexOf('\n');
100
101 // Skip header line
102 if (lineEnd == -1) {
103 _lastError = "Invalid CSV format";
104 return false;
105 }
106 lineStart = lineEnd + 1;
107
108 // Skip sample line (point -1)
109 lineEnd = csvData.indexOf('\n', lineStart);
110 if (lineEnd == -1) {
111 _lastError = "Missing sample line";
112 return false;
113 }
114 lineStart = lineEnd + 1;
115
116 // Clear existing alarms
117 _controller.clearConfiguredAlarms();
118
119 // Parse each data line
120 while (lineStart < csvData.length()) {
121 lineEnd = csvData.indexOf('\n', lineStart);
122 if (lineEnd == -1) lineEnd = csvData.length();
123
124 String line = csvData.substring(lineStart, lineEnd);
125 line.trim();
126
127 if (line.length() > 0) {
128 if (!_parseCSVLine(line)) {
129 return false;
130 }
131 }
132
133 lineStart = lineEnd + 1;
134 }
135
136 return true;
137}

147

153 if (alarm) {

156 return true;
157 }

163String CSVConfigManager::_escapeCSVField(const String& field) {
164 if (field.indexOf(',') >= 0 || field.indexOf('"') >= 0 || field.indexOf('\n') >= 0) {
165 String escaped = "\"";
166 for (int i = 0; i < field.length(); i++) {
167 if (field.charAt(i) == '"') {
168 escaped += "\"\"";
169 } else {
170 escaped += field.charAt(i);
171 }
172 }
173 escaped += "\"";
174 return escaped;
175 }
176 return field;

224bool CSVConfigManager::validatePointsCSV(const String& csvData) {
225 if (csvData.length() == 0) {
226 _lastError = "Empty CSV data";
227 return false;
228 }
229
230 // Check for required headers
231 String requiredHeaders[] = {
232 "PointAddress", "PointName", "PointType", "CurrentTemp", "MinTemp", "MaxTemp",
233 "LowTempThreshold", "HighTempThreshold", "SensorROM", "SensorBusNumber", // Changed header
234 "HIGH_TEMPERATURE", "LOW_TEMPERATURE", "SENSOR_ERROR", "SENSOR_DISCONNECTED"
235 };
236
237 int headerLine = csvData.indexOf('\n');
238 if (headerLine == -1) {
239 _lastError = "No header line found";
240 return false;
241 }
242
243 String header = csvData.substring(0, headerLine);
244 for (const String& reqHeader : requiredHeaders) {
245 if (header.indexOf(reqHeader) == -1) {
246 _lastError = "Missing required header: " + reqHeader;
247 return false;
248 }
249 }
250
251 return true;
252}

309bool CSVConfigManager::_parseCSVLine(const String& line) {
310 // Simple CSV parsing - split by commas
311 int fieldIndex = 0;
312 int startPos = 0;
313 String fields[14]; // 14 fields in our CSV format
314
315 for (int i = 0; i <= line.length(); i++) {
316 if (i == line.length() || line.charAt(i) == ',') {
317 if (fieldIndex < 14) {
318 fields[fieldIndex] = line.substring(startPos, i);
319 fields[fieldIndex].trim();
320 // Remove quotes if present
321 if (fields[fieldIndex].startsWith("\"") && fields[fieldIndex].endsWith("\"")) {
322 fields[fieldIndex] = fields[fieldIndex].substring(1, fields[fieldIndex].length() - 1);
323 }
324 }
325 fieldIndex++;
326 startPos = i + 1;
327 }
328 }
329
330 if (fieldIndex < 14) {
331 _lastError = "Insufficient fields in CSV line";
332 return false;
333 }
334
335 // Parse fields
336 int pointAddress = fields[0].toInt();
337
338 // Skip sample line (point -1)
339 if (pointAddress == -1) {
340 return true;
341 }
342
343 String pointName = fields[1];
344 String pointType = fields[2];
345 int16_t lowThreshold = fields[6].toInt();
346 int16_t highThreshold = fields[7].toInt();
347 String romString = fields[8];
348 String busNumberStr = fields[9]; // Changed from chipSelectStr
349
350 // Update measurement point
351 MeasurementPoint* point = _controller.getMeasurementPoint(pointAddress);
352 if (point) {
353 point->setName(pointName);
354 point->setLowAlarmThreshold(lowThreshold);
355 point->setHighAlarmThreshold(highThreshold);
356
357 // Bind sensor if specified
358 if (!romString.isEmpty()) {
359 _controller.bindSensorToPointByRom(romString, pointAddress);
360 } else if (!busNumberStr.isEmpty()) {
361 int busNumber = busNumberStr.toInt();
362 _controller.bindSensorToPointByBusNumber(busNumber, pointAddress); // Changed method name
363 }
364 }
365
366 // Parse alarms for each type (rest remains the same)
367 String alarmPriorities[4] = {fields[10], fields[11], fields[12], fields[13]};
368 AlarmType alarmTypes[4] = {
369 AlarmType::HIGH_TEMPERATURE,
370 AlarmType::LOW_TEMPERATURE,
371 AlarmType::SENSOR_ERROR,
372 AlarmType::SENSOR_DISCONNECTED
373 };
374
375 for (int i = 0; i < 4; i++) {
376 if (!alarmPriorities[i].isEmpty()) {
377 AlarmPriority priority = _parsePriority(alarmPriorities[i]);
378 if (!_controller.addAlarm(alarmTypes[i], pointAddress, priority)) {
379 _lastError = "Failed to add alarm for point " + String(pointAddress);
380 return false;
381 }
382 }
383 }
384
385 return true;
386}

389String CSVConfigManager::_getAlarmPriorityForPoint(int pointAddress, AlarmType alarmType) {
390 // Find alarm of specific type for this point
391 for (int i = 0; i < _controller.getAlarmCount(); i++) {
392 Alarm* alarm = _controller.getAlarmByIndex(i);
393 if (alarm && alarm->getPointAddress() == pointAddress && alarm->getType() == alarmType) {
