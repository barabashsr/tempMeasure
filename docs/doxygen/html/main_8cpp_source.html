251void loop() {
252 // Update time manager (handles RTC sync and NTP updates)
253 timeManager.update();
254
255 // Update configuration manager (handles web server and WiFi portal)
256 configManager->update();
257
258 // Update temperature controller (reads sensors and updates measurement points)
259 controller.update();
260
261 // Process any pending Modbus commands
262 if (modbusServer) {
263 modbusServer->processCommands();
264 }
265
266 // Periodic status output (when not in configuration portal mode)
267 static unsigned long lastPrintTime = 0;
268 if (!configManager->isPortalActive() &&
269 millis() - lastPrintTime > controller.getMeasurementPeriod()) {
270
271 // Uncomment for detailed status output:
272 // Serial.println("\nSystem Status:");
273 // Serial.println(controller.getSystemStatusJson());
274 // Serial.println(controller.getPointsJson());
275
276 lastPrintTime = millis();
277 }
278
279 // Update logger (handles periodic data logging and file management)
280 logger.update();
281
282 // Brief delay to prevent excessive CPU usage
283 delay(100);
284}
