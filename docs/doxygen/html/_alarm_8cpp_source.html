195bool Alarm::_checkCondition() {
196 if (!_source) {
197 Serial.println("Alarm: No source point");
198 return false;
199 }
200
201 int16_t currentTemp = _source->getCurrentTemp();
202 bool condition = false;
203
204 switch (_type) {
205 case AlarmType::HIGH_TEMPERATURE:
206 {
207 int16_t threshold = _source->getHighAlarmThreshold();
208 if (_stage == AlarmStage::CLEARED || _stage == AlarmStage::RESOLVED) {
209 // When cleared/resolved, need temperature to drop below (threshold - hysteresis) to stay cleared
210 // Return true if still above (threshold - hysteresis) = condition still exists
211 condition = currentTemp > (threshold - _hysteresis);
212 } else {
213 // Normal activation check: activate when >= threshold
214 condition = currentTemp >= threshold;
215 }
216 Serial.printf("HIGH_TEMP check: Point %d, Temp=%d, Threshold=%d, Hysteresis=%d, Stage=%s, Condition=%s\n",
217 _source->getAddress(), currentTemp, threshold, _hysteresis,
218 getStageString().c_str(), condition ? "TRUE" : "FALSE");
219 }
220 break;
221
222 case AlarmType::LOW_TEMPERATURE:
223 {
224 int16_t threshold = _source->getLowAlarmThreshold();
225 if (_stage == AlarmStage::CLEARED || _stage == AlarmStage::RESOLVED) {
226 // When cleared/resolved, need temperature to rise above (threshold + hysteresis) to stay cleared
227 // Return true if still below (threshold + hysteresis) = condition still exists
228 condition = currentTemp < (threshold + _hysteresis);
229 } else {
230 // Normal activation check: activate when <= threshold
231 condition = currentTemp <= threshold;
232 }
233 Serial.printf("LOW_TEMP check: Point %d, Temp=%d, Threshold=%d, Hysteresis=%d, Stage=%s, Condition=%s\n",
234 _source->getAddress(), currentTemp, threshold, _hysteresis,
235 getStageString().c_str(), condition ? "TRUE" : "FALSE");
236 }
237 break;
238
239 case AlarmType::SENSOR_ERROR:
240 condition = _source->getErrorStatus() != 0;
241 Serial.printf("SENSOR_ERROR check: Point %d, Error=%d, Condition=%s\n",
242 _source->getAddress(), _source->getErrorStatus(),
243 condition ? "TRUE" : "FALSE");
244 break;
245
246 case AlarmType::SENSOR_DISCONNECTED:
247 condition = _source->getBoundSensor() == nullptr;
248 Serial.printf("DISCONNECTED check: Point %d, Sensor=%p, Condition=%s\n",
249 _source->getAddress(), _source->getBoundSensor(),
250 condition ? "TRUE" : "FALSE");
251 break;
252
253 default:
254 Serial.println("Unknown alarm type");
255 return false;
256 }
257
258 return condition;
259}

478// Modify the updateCondition method to handle acknowledged timeout
479bool Alarm::updateCondition() {
480 if (!_source) {
481 Serial.println("Alarm updateCondition: No source");
482 return true;
483 }
484
485 bool conditionExists = _checkCondition();
486 AlarmStage oldStage = _stage;
487
488 Serial.printf("Alarm update: Point %d, Type=%s, Stage=%s, Condition=%s\n",
489 _source->getAddress(), getTypeString().c_str(),
490 getStageString().c_str(), conditionExists ? "EXISTS" : "CLEARED");
491
492 String source_ = "ALARM_" + String(_source->getAddress());
493 String baseDescription = getTypeString() + " alarm for point " +
494 String(_source->getAddress()) + " (" + _source->getName() + ")";
495
496 // Get current temperature and threshold for logging
497 int16_t currentTemp = _source->getCurrentTemp();
498 int16_t threshold = 0;
499
500 switch (_type) {
501 case AlarmType::HIGH_TEMPERATURE:
502 threshold = _source->getHighAlarmThreshold();
503 break;
504 case AlarmType::LOW_TEMPERATURE:
505 threshold = _source->getLowAlarmThreshold();
506 break;
507 case AlarmType::SENSOR_ERROR:
508 case AlarmType::SENSOR_DISCONNECTED:
509 threshold = 0; // Not applicable for these alarm types
510 break;
511 }
512
513 switch (_stage) {
514 case AlarmStage::NEW:
515 if (conditionExists) {
516 _stage = AlarmStage::ACTIVE;
517
518 // LOG: NEW -> ACTIVE
519 LoggerManager::error(source_, baseDescription + " activated");
520 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
521 getTypeString(), _getPriorityString(),
522 "NEW", "ACTIVE", currentTemp, threshold);
523
524 Serial.printf("Alarm %s: NEW -> ACTIVE\n", getTypeString().c_str());
525 } else {
526 resolve();
527
528 // LOG: NEW -> RESOLVED
529 LoggerManager::info(source_, baseDescription + " resolved before activation");
530 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
531 getTypeString(), _getPriorityString(),
532 "NEW", "RESOLVED", currentTemp, threshold);
533
534 Serial.printf("Alarm %s: NEW -> RESOLVED (condition cleared)\n", getTypeString().c_str());
535 }
536 break;
537
538 case AlarmStage::ACTIVE:
539 if (!conditionExists) {
540 clear();
541
542 // LOG: ACTIVE -> CLEARED
543 LoggerManager::info(source_, baseDescription + " condition cleared");
544 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
545 getTypeString(), _getPriorityString(),
546 "ACTIVE", "CLEARED", currentTemp, threshold);
547
548 Serial.printf("Alarm %s: ACTIVE -> CLEARED (condition no longer exists)\n", getTypeString().c_str());
549 }
550 break;
551
552 case AlarmStage::ACKNOWLEDGED:
553 if (!conditionExists) {
554 clear();
555
556 // LOG: ACKNOWLEDGED -> CLEARED
557 LoggerManager::info(source_, baseDescription + " condition cleared while acknowledged");
558 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
559 getTypeString(), _getPriorityString(),
560 "ACKNOWLEDGED", "CLEARED", currentTemp, threshold);
561
562 Serial.printf("Alarm %s: ACKNOWLEDGED -> CLEARED (condition no longer exists)\n", getTypeString().c_str());
563 } else if (isAcknowledgedDelayElapsed()) {
564 _stage = AlarmStage::ACTIVE;
565
566 // LOG: ACKNOWLEDGED -> ACTIVE (timeout)
567 LoggerManager::warning(source_, baseDescription + " acknowledgment timeout - returned to active");
568 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
569 getTypeString(), _getPriorityString(),
570 "ACKNOWLEDGED", "ACTIVE", currentTemp, threshold);
571
572 Serial.printf("Alarm %s: ACKNOWLEDGED -> ACTIVE (acknowledged delay elapsed)\n", getTypeString().c_str());
573 }
574 break;
575
576 case AlarmStage::CLEARED:
577 if (conditionExists) {
578 _stage = AlarmStage::ACTIVE;
579 _clearedTime = 0;
580
581 // LOG: CLEARED -> ACTIVE (condition returned)
582 LoggerManager::warning(source_, baseDescription + " condition returned");
583 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
584 getTypeString(), _getPriorityString(),
585 "CLEARED", "ACTIVE", currentTemp, threshold);
586
587 Serial.printf("Alarm %s: CLEARED -> ACTIVE (condition returned)\n", getTypeString().c_str());
588 } else if (isDelayElapsed()) {
589 resolve();
590
591 // LOG: CLEARED -> RESOLVED (delay elapsed)
592 LoggerManager::info(source_, baseDescription + " auto-resolved after delay");
593 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
594 getTypeString(), _getPriorityString(),
595 "CLEARED", "RESOLVED", currentTemp, threshold);
596
597 Serial.printf("Alarm %s: CLEARED -> RESOLVED (delay elapsed)\n", getTypeString().c_str());
598 }
599 break;
600
601 case AlarmStage::RESOLVED:
602 if (conditionExists) {
603 _stage = AlarmStage::ACTIVE;
604 _timestamp = millis();
605 _acknowledgedTime = 0;
606 _clearedTime = 0;
607
608 // LOG: RESOLVED -> ACTIVE (condition returned)
609 LoggerManager::error(source_, baseDescription + " reoccurred after resolution");
610 LoggerManager::logAlarmStateChange(_source->getAddress(), _source->getName(),
611 getTypeString(), _getPriorityString(),
612 "RESOLVED", "ACTIVE", currentTemp, threshold);
613
614 Serial.printf("Alarm %s: RESOLVED -> ACTIVE (condition returned)\n", getTypeString().c_str());
615 }
616 break;
617 }
618
619 if (oldStage != _stage) {
620 _updateMessage();
621 }
622
623 return true;
624}
