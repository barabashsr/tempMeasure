154ModbusMessage TempModbusServer::readHoldingRegistersWorker(ModbusMessage request) {
155 uint16_t address; // Requested register address
156 uint16_t words; // Requested number of registers
157 ModbusMessage response; // Response message to be sent back
158
159 // Get request values
160 request.get(2, address);
161 request.get(4, words);
162
163 // LOG: Read request received
164 LoggerManager::info("MODBUS_READ",
165 "Read request - Address: " + String(address) +
166 ", Count: " + String(words) +
167 ", Client ID: " + String(request.getServerID()));
168
169 // Check if address and word count are valid
170 if (words > 0 && words <= 125) { // Max 125 registers per request as per Modbus spec
171 // Set up response header
172 response.add(request.getServerID(), request.getFunctionCode(), (uint8_t)(words * 2));
173
174 // Add requested register values to response
175 bool allValid = true;
176 String readValues = "";
177
178 for (uint16_t i = 0; i < words; i++) {
179 uint16_t regValue = registerMapPtr->readHoldingRegister(address + i);
180
181 // Check if register read was successful - 0xFFFF is often used as an error indicator
182 if (regValue != 0xFFFF) {
183 response.add(regValue);
184 if (i < 5) { // Log first 5 values to avoid spam
185 readValues += String(address + i) + ":" + String(regValue) + " ";
186 } else if (i == 5) {
187 readValues += "...";
188 }
189 } else {
190 allValid = false;
191 LoggerManager::error("MODBUS_READ",
192 "Failed to read register " + String(address + i));
193 break;
194 }
195 }
196
197 // If any register was invalid, return error response
198 if (!allValid) {
199 response.clear();
200 response.setError(request.getServerID(), request.getFunctionCode(), ILLEGAL_DATA_ADDRESS);
201 LoggerManager::error("MODBUS_READ",
202 "Read failed - Invalid register address range: " +
203 String(address) + "-" + String(address + words - 1));
204 } else {
205 LoggerManager::info("MODBUS_READ",
206 "Read successful - " + String(words) + " registers: " + readValues);
207 }
208 } else {
209 // Invalid word count, return error response
210 response.setError(request.getServerID(), request.getFunctionCode(), ILLEGAL_DATA_ADDRESS);
211 LoggerManager::error("MODBUS_READ",
212 "Read failed - Invalid word count: " + String(words) + " (max 125)");
213 }
214
215 return response;
216}

315ModbusMessage TempModbusServer::writeMultipleRegistersWorker(ModbusMessage request) {
316 uint16_t address; // Starting register address
317 uint16_t words; // Number of registers to write
318 uint8_t bytesCount; // Number of data bytes in request
319 ModbusMessage response; // Response message to be sent back
320
321 // Get request values
322 request.get(2, address);
323 request.get(4, words);
324 request.get(6, bytesCount);
325
326 // LOG: Multiple write request received
327 LoggerManager::info("MODBUS_WRITE",
328 "Write multiple registers - Start: " + String(address) +
329 ", Count: " + String(words) +
330 ", Bytes: " + String(bytesCount) +
331 ", Client ID: " + String(request.getServerID()));
332
333 // Check if word count is valid
334 if (words > 0 && words <= 123 && bytesCount == words * 2) { // Max 123 registers per request
335 bool allWritten = true;
336 String writeValues = "";
337
338 // Write each register value
339 for (uint16_t i = 0; i < words; i++) {
340 uint16_t value;
341 request.get(7 + i * 2, value);
342
343 if (registerMapPtr->writeHoldingRegister(address + i, value)) {
344 if (i < 5) { // Log first 5 values to avoid spam
345 writeValues += String(address + i) + ":" + String(value) + " ";
346 } else if (i == 5) {
347 writeValues += "...";
348 }
349 } else {
350 allWritten = false;
351 LoggerManager::error("MODBUS_WRITE",
352 "Failed to write register " + String(address + i) + " = " + String(value));
353 break;
354 }
355 }
356
357 if (allWritten) {
358 // Success - create response with address and word count
359 response.add(request.getServerID(), request.getFunctionCode());
360 response.add(address);
361 response.add(words);
362 LoggerManager::info("MODBUS_WRITE",
363 "Multiple write successful - " + String(words) + " registers: " + writeValues);
364 } else {
365 // Failed to write at least one register
366 response.setError(request.getServerID(), request.getFunctionCode(), ILLEGAL_DATA_ADDRESS);
367 LoggerManager::error("MODBUS_WRITE",
368 "Multiple write failed - Error writing to register range: " +
369 String(address) + "-" + String(address + words - 1));
370 }
371 } else {
372 // Invalid word count or byte count
373 response.setError(request.getServerID(), request.getFunctionCode(), ILLEGAL_DATA_ADDRESS);
374 LoggerManager::error("MODBUS_WRITE",
375 "Multiple write failed - Invalid parameters: Words=" + String(words) +
376 ", Bytes=" + String(bytesCount) + " (expected " + String(words * 2) + ")");
377 }
378
379 return response;
