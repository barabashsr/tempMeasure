56uint16_t RegisterMap::readHoldingRegister(uint16_t address) {
57 if (!isValidAddress(address)) return 0xFFFF;
58
59 if (address == DEVICE_ID_REG) return deviceId;
60 if (address == FIRMWARE_VERSION_REG) return firmwareVersion;
61 if (address == NUM_DS18B20_REG) return numActiveDS18B20;
62 if (address == NUM_PT1000_REG) return numActivePT1000;
63 if (address >= DEVICE_STATUS_START_REG && address <= DEVICE_STATUS_END_REG)
64 return deviceStatus[address - DEVICE_STATUS_START_REG];
65
66 // DS18B20 and PT1000 share the same arrays, just different index offsets
67 if (address >= CURRENT_TEMP_DS18B20_START_REG && address <= CURRENT_TEMP_PT1000_END_REG)
68 return currentTemps[address - CURRENT_TEMP_DS18B20_START_REG];
69 if (address >= MIN_TEMP_DS18B20_START_REG && address <= MIN_TEMP_PT1000_END_REG)
70 return minTemps[address - MIN_TEMP_DS18B20_START_REG];
71 if (address >= MAX_TEMP_DS18B20_START_REG && address <= MAX_TEMP_PT1000_END_REG)
72 return maxTemps[address - MAX_TEMP_DS18B20_START_REG];
73 if (address >= ALARM_STATUS_DS18B20_START_REG && address <= ALARM_STATUS_PT1000_END_REG)
74 return alarmStatus[address - ALARM_STATUS_DS18B20_START_REG];
75 if (address >= ERROR_STATUS_DS18B20_START_REG && address <= ERROR_STATUS_PT1000_END_REG)
76 return errorStatus[address - ERROR_STATUS_DS18B20_START_REG];
77 if (address >= LOW_ALARM_DS18B20_START_REG && address <= LOW_ALARM_PT1000_END_REG)
78 return lowAlarmThresholds[address - LOW_ALARM_DS18B20_START_REG];
79 if (address >= HIGH_ALARM_DS18B20_START_REG && address <= HIGH_ALARM_PT1000_END_REG)
80 return highAlarmThresholds[address - HIGH_ALARM_DS18B20_START_REG];
81
82 return 0xFFFF;
83}
