20RegisterMap::RegisterMap() {
21 deviceId = 1000;
22 firmwareVersion = 0x0100;
23 numActiveDS18B20 = 0;
24 numActivePT1000 = 0;
25 commandRegister = 0;
26 commandPending = false;
27
28 for (int i = 0; i < 7; i++) deviceStatus[i] = 0;
29 for (int i = 0; i < 60; i++) {
30 currentTemps[i] = 0;
31 minTemps[i] = 32767;
32 maxTemps[i] = -32768;
33 alarmStatus[i] = 0;
34 errorStatus[i] = 0;
35 lowAlarmThresholds[i] = -10;
36 highAlarmThresholds[i] = 50;
37 // Initialize alarm config with all alarms disabled, medium priority
38 alarmConfig[i] = (1 << ALARM_CONFIG_LOW_PRIORITY_SHIFT) |
39 (1 << ALARM_CONFIG_HIGH_PRIORITY_SHIFT) |
40 (1 << ALARM_CONFIG_ERROR_PRIORITY_SHIFT);
41 }
42
43 // Initialize relay control (all auto mode)
44 for (int i = 0; i < 6; i++) {
45 relayControl[i] = 0;
46 }
47
48 // Initialize relay status registers
49 for (int i = 0; i < 3; i++) {
50 relayStatus[i] = 0;
51 }
52
53 // Initialize hysteresis to 5 degrees
54 for (int i = 0; i < 20; i++) {
55 hysteresis[i] = 50; // 5.0 degrees in 0.1 degree units
56 }
57}

59bool RegisterMap::isValidAddress(uint16_t address) {
60 if (address <= DEVICE_STATUS_END_REG) return true;
61 if (address >= RELAY_STATUS_REG_START && address <= RELAY_STATUS_REG_END) return true;
62 if (address >= CURRENT_TEMP_DS18B20_START_REG && address <= CURRENT_TEMP_PT1000_END_REG) return true;
63 if (address >= MIN_TEMP_DS18B20_START_REG && address <= MIN_TEMP_PT1000_END_REG) return true;
64 if (address >= MAX_TEMP_DS18B20_START_REG && address <= MAX_TEMP_PT1000_END_REG) return true;
65 if (address >= ALARM_STATUS_DS18B20_START_REG && address <= ALARM_STATUS_PT1000_END_REG) return true;
66 if (address >= ERROR_STATUS_DS18B20_START_REG && address <= ERROR_STATUS_PT1000_END_REG) return true;
67 if (address >= LOW_ALARM_DS18B20_START_REG && address <= LOW_ALARM_PT1000_END_REG) return true;
68 if (address >= HIGH_ALARM_DS18B20_START_REG && address <= HIGH_ALARM_PT1000_END_REG) return true;
69 if (address >= ALARM_CONFIG_DS18B20_START_REG && address <= ALARM_CONFIG_PT1000_END_REG) return true;
70 if (address >= RELAY_CONTROL_START_REG && address <= RELAY_CONTROL_END_REG) return true;
71 if (address >= HYSTERESIS_START_REG && address <= HYSTERESIS_END_REG) return true;
72 if (address == COMMAND_REG) return true;
73 return false;
74}

76bool RegisterMap::isReadOnlyRegister(uint16_t address) {
77 // Writable registers
78 if (address >= LOW_ALARM_DS18B20_START_REG && address <= LOW_ALARM_PT1000_END_REG) return false;
79 if (address >= HIGH_ALARM_DS18B20_START_REG && address <= HIGH_ALARM_PT1000_END_REG) return false;

81 if (address >= RELAY_CONTROL_START_REG && address <= RELAY_CONTROL_START_REG + 2) return false; // Only control, not status
82 if (address >= HYSTERESIS_START_REG && address <= HYSTERESIS_END_REG) return false;
83 if (address == COMMAND_REG) return false;

87uint16_t RegisterMap::readHoldingRegister(uint16_t address) {
88 if (!isValidAddress(address)) return 0xFFFF;
89
90 if (address == DEVICE_ID_REG) return deviceId;
91 if (address == FIRMWARE_VERSION_REG) return firmwareVersion;
92 if (address == NUM_DS18B20_REG) return numActiveDS18B20;
93 if (address == NUM_PT1000_REG) return numActivePT1000;
94 if (address >= DEVICE_STATUS_START_REG && address <= DEVICE_STATUS_END_REG)
95 return deviceStatus[address - DEVICE_STATUS_START_REG];
96 if (address >= RELAY_STATUS_REG_START && address <= RELAY_STATUS_REG_END)
97 return relayStatus[address - RELAY_STATUS_REG_START];
98
99 // DS18B20 and PT1000 share the same arrays, just different index offsets
100 if (address >= CURRENT_TEMP_DS18B20_START_REG && address <= CURRENT_TEMP_PT1000_END_REG)
101 return currentTemps[address - CURRENT_TEMP_DS18B20_START_REG];
102 if (address >= MIN_TEMP_DS18B20_START_REG && address <= MIN_TEMP_PT1000_END_REG)
103 return minTemps[address - MIN_TEMP_DS18B20_START_REG];
104 if (address >= MAX_TEMP_DS18B20_START_REG && address <= MAX_TEMP_PT1000_END_REG)
105 return maxTemps[address - MAX_TEMP_DS18B20_START_REG];
106 if (address >= ALARM_STATUS_DS18B20_START_REG && address <= ALARM_STATUS_PT1000_END_REG)
107 return alarmStatus[address - ALARM_STATUS_DS18B20_START_REG];
108 if (address >= ERROR_STATUS_DS18B20_START_REG && address <= ERROR_STATUS_PT1000_END_REG)
109 return errorStatus[address - ERROR_STATUS_DS18B20_START_REG];
110 if (address >= LOW_ALARM_DS18B20_START_REG && address <= LOW_ALARM_PT1000_END_REG)
111 return lowAlarmThresholds[address - LOW_ALARM_DS18B20_START_REG];
112 if (address >= HIGH_ALARM_DS18B20_START_REG && address <= HIGH_ALARM_PT1000_END_REG)
113 return highAlarmThresholds[address - HIGH_ALARM_DS18B20_START_REG];
114
115 // Alarm configuration registers
116 if (address >= ALARM_CONFIG_DS18B20_START_REG && address <= ALARM_CONFIG_PT1000_END_REG)
117 return alarmConfig[address - ALARM_CONFIG_DS18B20_START_REG];
118
119 // Relay control and status registers
120 if (address >= RELAY_CONTROL_START_REG && address <= RELAY_CONTROL_END_REG)
121 return relayControl[address - RELAY_CONTROL_START_REG];
122
123 // Hysteresis registers
124 if (address >= HYSTERESIS_START_REG && address <= HYSTERESIS_END_REG)
125 return hysteresis[address - HYSTERESIS_START_REG];
126
127 // Command register
128 if (address == COMMAND_REG)
129 return commandRegister;
130
131 return 0xFFFF;
132}

134bool RegisterMap::writeHoldingRegister(uint16_t address, uint16_t value) {
135 if (!isValidAddress(address)) return false;
136 if (isReadOnlyRegister(address)) return false;
137
138 // Alarm thresholds
139 if (address >= LOW_ALARM_DS18B20_START_REG && address <= LOW_ALARM_PT1000_END_REG) {
140 lowAlarmThresholds[address - LOW_ALARM_DS18B20_START_REG] = static_cast<int16_t>(value);
141 return true;
142 }
143 if (address >= HIGH_ALARM_DS18B20_START_REG && address <= HIGH_ALARM_PT1000_END_REG) {
144 highAlarmThresholds[address - HIGH_ALARM_DS18B20_START_REG] = static_cast<int16_t>(value);
145 return true;
146 }
147
148 // Alarm configuration
149 if (address >= ALARM_CONFIG_DS18B20_START_REG && address <= ALARM_CONFIG_PT1000_END_REG) {
150 alarmConfig[address - ALARM_CONFIG_DS18B20_START_REG] = value;
151 return true;
152 }
153
154 // Relay control (only control registers, not status)
155 if (address >= RELAY_CONTROL_START_REG && address <= RELAY_CONTROL_START_REG + 2) {
156 relayControl[address - RELAY_CONTROL_START_REG] = value;
157 return true;
158 }
159
160 // Hysteresis
161 if (address >= HYSTERESIS_START_REG && address <= HYSTERESIS_END_REG) {
162 hysteresis[address - HYSTERESIS_START_REG] = value;
163 return true;
164 }
165
166 // Command register
167 if (address == COMMAND_REG) {
168 commandRegister = value;
169 commandPending = true;
170 return true;
171 }
172
173 return false;
174}

240void RegisterMap::updateRelayStatusRegister(uint8_t relayIndex, bool commandedState, bool actualState) {
241 if (relayIndex >= 3) return;
242
243 // Relay status register format:
244 // Bit 0: Commanded state (what system wants relay to be)
245 // Bit 1: Actual state (current hardware state)
246 uint16_t statusValue = 0;
247 if (commandedState) statusValue |= 0x0001;
248 if (actualState) statusValue |= 0x0002;
249
250 relayStatus[relayIndex] = statusValue;
251}
