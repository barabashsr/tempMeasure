291void TemperatureController::_checkPointForAlarms(MeasurementPoint* point) {
292 if (!point || !point->getBoundSensor()) return;
293
294 // Check for high temperature alarm
295 if (point->getCurrentTemp() >= point->getHighAlarmThreshold()) {
296 if (!_hasAlarmForPoint(point, AlarmType::HIGH_TEMPERATURE)) {
297 createAlarm(AlarmType::HIGH_TEMPERATURE, point, AlarmPriority::PRIORITY_HIGH);
298 }
299 }
300
301 // Check for low temperature alarm
302 if (point->getCurrentTemp() <= point->getLowAlarmThreshold()) {
303 if (!_hasAlarmForPoint(point, AlarmType::LOW_TEMPERATURE)) {
304 createAlarm(AlarmType::LOW_TEMPERATURE, point, AlarmPriority::PRIORITY_MEDIUM);
305 }
306 }
307
308 // Check for sensor error
309 if (point->getErrorStatus() != 0) {
310 if (!_hasAlarmForPoint(point, AlarmType::SENSOR_ERROR)) {
311 createAlarm(AlarmType::SENSOR_ERROR, point, AlarmPriority::PRIORITY_HIGH);
312 }

554void TemperatureController::handleAlarmOutputs() {
555 // Get alarm counts by priority and stage for precise control
556 int criticalActive = getAlarmCount(AlarmPriority::PRIORITY_CRITICAL, AlarmStage::ACTIVE);
557 int criticalAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_CRITICAL, AlarmStage::ACKNOWLEDGED);
558 int highActive = getAlarmCount(AlarmPriority::PRIORITY_HIGH, AlarmStage::ACTIVE);
559 int highAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_HIGH, AlarmStage::ACKNOWLEDGED);
560 int mediumActive = getAlarmCount(AlarmPriority::PRIORITY_MEDIUM, AlarmStage::ACTIVE);
561 int mediumAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_MEDIUM, AlarmStage::ACKNOWLEDGED);
562 int lowActive = getAlarmCount(AlarmPriority::PRIORITY_LOW, AlarmStage::ACTIVE);
563 int lowAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_LOW, AlarmStage::ACKNOWLEDGED);
564
565 // Determine highest priority alarm state for output control
566 bool hasCritical = (criticalActive + criticalAcknowledged) > 0;
567 bool hasHigh = (highActive + highAcknowledged) > 0;
568 bool hasMedium = (mediumActive + mediumAcknowledged) > 0;
569 bool hasLow = (lowActive + lowAcknowledged) > 0;
570
571 // Determine if alarms are acknowledged
572 bool criticalAcknowledgedOnly = (criticalActive == 0 && criticalAcknowledged > 0);
573 bool highAcknowledgedOnly = (highActive == 0 && highAcknowledged > 0);
574 bool mediumAcknowledgedOnly = (mediumActive == 0 && mediumAcknowledged > 0);
575 bool lowAcknowledgedOnly = (lowActive == 0 && lowAcknowledged > 0);
576
577 // Apply alarm notification logic based on priority and acknowledgment state
578 // Following the specification table from PLANNING_RESULTS.md
579
580 // Relay1 (Siren) - on for ANY active alarm of ANY priority
581 bool relay1State = (criticalActive + highActive + mediumActive + lowActive) > 0;
582 bool relay2State = false; // Flash beacon
583 bool redLedState = false;
584 bool yellowLedState = false;
585 bool blueLedState = false;
586
587 // Track what should be blinking to avoid unnecessary restarts
588 static bool relay2WasBlinking = false;
589 static bool blueLedWasBlinking = false;
590 static unsigned long relay2LastOn = 0, relay2LastOff = 0;
591 static unsigned long blueLastOn = 0, blueLastOff = 0;
592
593 // Determine new blinking requirements
594 bool relay2ShouldBlink = false;
595 bool blueLedShouldBlink = false;
596 unsigned long relay2OnTime = 0, relay2OffTime = 0;
597 unsigned long blueOnTime = 0, blueOffTime = 0;
598
599 if (hasCritical) {
600 if (criticalAcknowledgedOnly) {
601 // CRITICAL acknowledged: Beacon ON (constant)
602 relay2State = true;
603 } else {
604 // CRITICAL active: Siren ON + Beacon ON (constant)
605 // relay1State already set based on ANY active alarm
606 relay2State = true;
607 }
608 // Red LED solid for critical alarms
609 redLedState = true;
610 } else if (hasHigh) {
611 if (highAcknowledgedOnly) {
612 // HIGH acknowledged: Beacon ON (blink 2s on/30s off)
613 relay2ShouldBlink = true;
614 relay2OnTime = 2000;
615 relay2OffTime = 30000;
616 } else {
617 // HIGH active: Beacon ON (constant)
618 relay2State = true;
619 }
620 // Yellow LED solid for high priority
621 yellowLedState = true;
622 } else if (hasMedium) {
623 if (!mediumAcknowledgedOnly) {
624 // MEDIUM active: Beacon ON (blink 2s on/30s off)
625 relay2ShouldBlink = true;
626 relay2OnTime = 2000;
627 relay2OffTime = 30000;
628 }
629 // MEDIUM acknowledged: Beacon OFF (no relay action)
630 // Blue LED solid for medium priority
631 blueLedState = true;
632 } else if (hasLow) {
633 // LOW priority: No relay action (as per specification)
634 // Blue LED blinking for low priority alarms
635 if (!lowAcknowledgedOnly) {
636 blueLedShouldBlink = true;
637 blueOnTime = 200;
638 blueOffTime = 2000; // 200ms on, 2000ms off - short flash
639 }
640 }
641
642 // Green LED is ON when there are no alarms (system OK)
643 bool greenLedState = !hasCritical && !hasHigh && !hasMedium && !hasLow;
644
645 // Update relay2 blinking only if state changed
646 if (relay2ShouldBlink != relay2WasBlinking ||
647 (relay2ShouldBlink && (relay2OnTime != relay2LastOn || relay2OffTime != relay2LastOff))) {
648 if (relay2ShouldBlink) {
649 indicator.startBlinking("Relay2", relay2OnTime, relay2OffTime);
650 } else {
651 indicator.stopBlinking("Relay2");
652 }
653 relay2WasBlinking = relay2ShouldBlink;
654 relay2LastOn = relay2OnTime;
655 relay2LastOff = relay2OffTime;
656 }
657
658 // Update blue LED blinking only if state changed
659 if (blueLedShouldBlink != blueLedWasBlinking ||
660 (blueLedShouldBlink && (blueOnTime != blueLastOn || blueOffTime != blueLastOff))) {
661 if (blueLedShouldBlink) {
662 indicator.startBlinking("BlueLED", blueOnTime, blueOffTime);
663 } else {
664 indicator.stopBlinking("BlueLED");
665 }
666 blueLedWasBlinking = blueLedShouldBlink;
667 blueLastOn = blueOnTime;
668 blueLastOff = blueOffTime;
669 }
670
671 // Log alarm summary periodically
672 static unsigned long lastSummaryLog = 0;
673 unsigned long now = millis();
674 if (now - lastSummaryLog > 30000 || // Log every 30 seconds
675 (hasCritical || hasHigh || hasMedium) && now - lastSummaryLog > 5000) { // Or every 5 seconds if significant alarms
676
677 String summary = "Alarm summary - Critical: " + String(criticalActive) + "/" + String(criticalAcknowledged) +
678 ", High: " + String(highActive) + "/" + String(highAcknowledged) +
679 ", Medium: " + String(mediumActive) + "/" + String(mediumAcknowledged) +
680 ", Low: " + String(lowActive) + "/" + String(lowAcknowledged);
681 LoggerManager::info("ALARM_OUTPUT", summary);
682 lastSummaryLog = now;
683 }
684
685 // Apply relay control modes
686 bool finalRelay1State = relay1State;
687 bool finalRelay2State = relay2State;
688 bool finalRelay3State = _relay3State; // Default to current state
689
690 // Handle Relay 1 control mode
691 if (_relay1Mode == RelayControlMode::FORCE_OFF) {
692 finalRelay1State = false;
693 } else if (_relay1Mode == RelayControlMode::FORCE_ON) {
694 finalRelay1State = true;
695 }
696 // else AUTO mode - use alarm-based state
697
698 // Handle Relay 2 control mode (stop blinking if forced)
699 if (_relay2Mode == RelayControlMode::FORCE_OFF) {
700 indicator.stopBlinking("Relay2");
701 finalRelay2State = false;
702 } else if (_relay2Mode == RelayControlMode::FORCE_ON) {
703 indicator.stopBlinking("Relay2");
704 finalRelay2State = true;
705 }
706 // else AUTO mode - use alarm-based state (including blinking)
707
708 // Handle Relay 3 control mode (Modbus-only, no alarm control)
709 if (_relay3Mode == RelayControlMode::FORCE_OFF) {
710 finalRelay3State = false;
711 } else if (_relay3Mode == RelayControlMode::FORCE_ON) {
712 finalRelay3State = true;
713 }
714 // Note: Relay3 has no AUTO behavior - it's Modbus-only
715
716 // Update relay states
717 if (!indicator.isBlinking("Relay1") && finalRelay1State != _relay1State) {
718 LoggerManager::info("INDICATION",
719 "Relay1 (Siren) state change: " + String(_relay1State ? "ON" : "OFF") +
720 " -> " + String(finalRelay1State ? "ON" : "OFF") +
721 " (Mode: " + String(static_cast<int>(_relay1Mode)) + ")");
722 indicator.writePort("Relay1", finalRelay1State);
723 _relay1State = finalRelay1State;
724 }

1928 });
