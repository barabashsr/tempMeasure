237void TemperatureController::_checkPointForAlarms(MeasurementPoint* point) {
238 if (!point || !point->getBoundSensor()) return;
239
240 // Check for high temperature alarm
241 if (point->getCurrentTemp() >= point->getHighAlarmThreshold()) {
242 if (!_hasAlarmForPoint(point, AlarmType::HIGH_TEMPERATURE)) {
243 createAlarm(AlarmType::HIGH_TEMPERATURE, point, AlarmPriority::PRIORITY_HIGH);
244 }
245 }
246
247 // Check for low temperature alarm
248 if (point->getCurrentTemp() <= point->getLowAlarmThreshold()) {
249 if (!_hasAlarmForPoint(point, AlarmType::LOW_TEMPERATURE)) {
250 createAlarm(AlarmType::LOW_TEMPERATURE, point, AlarmPriority::PRIORITY_MEDIUM);
251 }
252 }
253
254 // Check for sensor error
255 if (point->getErrorStatus() != 0) {
256 if (!_hasAlarmForPoint(point, AlarmType::SENSOR_ERROR)) {
257 createAlarm(AlarmType::SENSOR_ERROR, point, AlarmPriority::PRIORITY_HIGH);
258 }
259 }

275void TemperatureController::createAlarm(AlarmType type, MeasurementPoint* source, AlarmPriority priority) {
276 // Check if this alarm already exists in configured alarms
277 String configKey = "alarm_" + String(source->getAddress()) + "_" + String(static_cast<int>(type));
278
279 for (auto alarm : _configuredAlarms) {
280 if (alarm->getConfigKey() == configKey) {
281 // Alarm already exists, just enable it if it's disabled
282 if (!alarm->isEnabled()) {
283 alarm->setEnabled(true);
284 alarm->setStage(AlarmStage::NEW); // Reset stage
285 }
286 return;
287 }
288 }
289
290 // Create new alarm and add to configured alarms
291 Alarm* newAlarm = new Alarm(type, source, priority);
292 newAlarm->setConfigKey(configKey);
293 _configuredAlarms.push_back(newAlarm);
294
295 // Sort alarms by priority
296 std::sort(_configuredAlarms.begin(), _configuredAlarms.end(), AlarmComparator());
297}

429void TemperatureController::handleAlarmOutputs() {
430 // Get alarm counts by priority and stage for precise control
431 int criticalActive = getAlarmCount(AlarmPriority::PRIORITY_CRITICAL, AlarmStage::ACTIVE);
432 int criticalAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_CRITICAL, AlarmStage::ACKNOWLEDGED);
433 int highActive = getAlarmCount(AlarmPriority::PRIORITY_HIGH, AlarmStage::ACTIVE);
434 int highAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_HIGH, AlarmStage::ACKNOWLEDGED);
435 int mediumActive = getAlarmCount(AlarmPriority::PRIORITY_MEDIUM, AlarmStage::ACTIVE);
436 int mediumAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_MEDIUM, AlarmStage::ACKNOWLEDGED);
437 int lowActive = getAlarmCount(AlarmPriority::PRIORITY_LOW, AlarmStage::ACTIVE);
438 int lowAcknowledged = getAlarmCount(AlarmPriority::PRIORITY_LOW, AlarmStage::ACKNOWLEDGED);
439
440 // Determine highest priority alarm state for output control
441 bool hasCritical = (criticalActive + criticalAcknowledged) > 0;
442 bool hasHigh = (highActive + highAcknowledged) > 0;
443 bool hasMedium = (mediumActive + mediumAcknowledged) > 0;
444 bool hasLow = (lowActive + lowAcknowledged) > 0;
445
446 // Determine if alarms are acknowledged
447 bool criticalAcknowledgedOnly = (criticalActive == 0 && criticalAcknowledged > 0);
448 bool highAcknowledgedOnly = (highActive == 0 && highAcknowledged > 0);
449 bool mediumAcknowledgedOnly = (mediumActive == 0 && mediumAcknowledged > 0);
450
451 // Apply alarm notification logic based on priority and acknowledgment state
452 // Following the specification table from PLANNING_RESULTS.md
453
454 bool relay1State = false; // Siren
455 bool relay2State = false; // Flash beacon
456 bool redLedState = false;
457 bool yellowLedState = false;
458 bool blueLedState = false;
459
460 // Stop all blinking first - we'll restart as needed
461 indicator.stopBlinking("Relay2");
462 indicator.stopBlinking("YellowLED");
463
464 if (hasCritical) {
465 if (criticalAcknowledgedOnly) {
466 // CRITICAL acknowledged: Beacon ON (constant)
467 relay2State = true;
468 } else {
469 // CRITICAL active: Siren ON + Beacon ON (constant)
470 relay1State = true;
471 relay2State = true;
472 }
473 // Red LED always on for critical alarms
474 redLedState = true;
475 yellowLedState = true; // Both LEDs for critical
476 } else if (hasHigh) {
477 if (highAcknowledgedOnly) {
478 // HIGH acknowledged: Beacon ON (blink 2s on/30s off)
479 indicator.startBlinking("Relay2", 2000, 30000);
480 } else {
481 // HIGH active: Beacon ON (constant)
482 relay2State = true;
483 }
484 // Red LED for high priority (when not acknowledged)
485 redLedState = !highAcknowledgedOnly;
486 yellowLedState = true;
487 } else if (hasMedium) {
488 if (!mediumAcknowledgedOnly) {
489 // MEDIUM active: Beacon ON (blink 2s on/30s off)
490 indicator.startBlinking("Relay2", 2000, 30000);
491 }
492 // MEDIUM acknowledged: Beacon OFF (no relay action)
493 // Yellow LED blinking for medium priority
494 if (!mediumAcknowledgedOnly) {
495 indicator.startBlinking("YellowLED", 2000, 30000);
496 }
497 } else if (hasLow) {
498 // LOW priority: No relay action (as per specification)
499 // No LED indication for low priority
500 }
501
502 // Log alarm summary periodically
503 static unsigned long lastSummaryLog = 0;
504 unsigned long now = millis();
505 if (now - lastSummaryLog > 30000 || // Log every 30 seconds
506 (hasCritical || hasHigh || hasMedium) && now - lastSummaryLog > 5000) { // Or every 5 seconds if significant alarms
507
508 String summary = "Alarm summary - Critical: " + String(criticalActive) + "/" + String(criticalAcknowledged) +
509 ", High: " + String(highActive) + "/" + String(highAcknowledged) +
510 ", Medium: " + String(mediumActive) + "/" + String(mediumAcknowledged) +
511 ", Low: " + String(lowActive) + "/" + String(lowAcknowledged);
512 LoggerManager::info("ALARM_OUTPUT", summary);
513 lastSummaryLog = now;
514 }
515
516 // Update relay states (only if not blinking)
517 if (!indicator.isBlinking("Relay1") && relay1State != _relay1State) {
518 LoggerManager::info("INDICATION",
519 "Relay1 (Siren) state change: " + String(_relay1State ? "ON" : "OFF") +
520 " -> " + String(relay1State ? "ON" : "OFF"));
521 indicator.writePort("Relay1", relay1State);
522 _relay1State = relay1State;
523 }
524
525 if (!indicator.isBlinking("Relay2") && relay2State != _relay2State) {
526 LoggerManager::info("INDICATION",
527 "Relay2 (Beacon) state change: " + String(_relay2State ? "ON" : "OFF") +
528 " -> " + String(relay2State ? "ON" : "OFF"));
529 indicator.writePort("Relay2", relay2State);
530 _relay2State = relay2State;
531 }
532
533 // Update LED states (only if not blinking)
534 if (redLedState != _redLedState) {
535 LoggerManager::info("INDICATION",
536 "Red LED state change: " + String(_redLedState ? "ON" : "OFF") +
537 " -> " + String(redLedState ? "ON" : "OFF"));
538 indicator.writePort("RedLED", redLedState);
539 _redLedState = redLedState;
540 }
541
542 if (!indicator.isBlinking("YellowLED") && yellowLedState != _yellowLedState) {
543 LoggerManager::info("INDICATION",
544 "Yellow LED state change: " + String(_yellowLedState ? "ON" : "OFF") +
545 " -> " + String(yellowLedState ? "ON" : "OFF"));
546 indicator.writePort("YellowLED", yellowLedState);
547 _yellowLedState = yellowLedState;
548 }
549
550 if (blueLedState != _blueLedState) {
551 LoggerManager::info("INDICATION",
552 "Blue LED state change: " + String(_blueLedState ? "ON" : "OFF") +
553 " -> " + String(blueLedState ? "ON" : "OFF"));
554 indicator.writePort("BlueLED", blueLedState);
555 _blueLedState = blueLedState;
556 }
557}

1312 MeasurementPoint* point = getMeasurementPoint(pointAddress);
1313 if (!point) return false;
1314
1315 // Check if alarm already exists
1316 String configKey = "alarm_" + String(pointAddress) + "_" + String(static_cast<int>(type));
1317
1318 for (auto alarm : _configuredAlarms) {
1319 if (alarm->getConfigKey() == configKey) {
1320 // Update existing
1321 alarm->setPriority(priority);
1322 alarm->setEnabled(true);
1323 return true;
1324 }
1325 }
1326
1327 // Create new alarm
1328 Alarm* newAlarm = new Alarm(type, point, priority);
1329 if (newAlarm) {
1330 unsigned long delay;
1331 switch (priority) {
1332 case AlarmPriority::PRIORITY_CRITICAL:
1333 delay = _acknowledgedDelayCritical;
1334 break;
1335 case AlarmPriority::PRIORITY_HIGH:
1336 delay = _acknowledgedDelayHigh;
1337 break;

1339 delay = _acknowledgedDelayMedium;
1340 break;
1341 case AlarmPriority::PRIORITY_LOW:
1342 delay = _acknowledgedDelayLow;
1343 break;
1344 default:
1345 delay = _acknowledgedDelayMedium;
1346 break;
1347 }
1348 newAlarm->setAcknowledgedDelay(delay);
1349 }
1350 newAlarm->setConfigKey(configKey);
1351 _configuredAlarms.push_back(newAlarm);
1352
1353 Serial.printf("Added alarm configuration: %s\n", configKey.c_str());
1354 if (newAlarm) {
1355 MeasurementPoint* point = getMeasurementPoint(pointAddress);
1356 String pointName = point ? point->getName() : "Unknown";
1357 LoggerManager::info("ALARM_CONFIG",
1358 "Added " + newAlarm->getTypeString() + " alarm for point " +
1359 String(pointAddress) + " (" + pointName + ") with priority " +
1360 _getPriorityString(priority));

1362 LoggerManager::warning("ALARM_CONFIG",
1363 "Failed to add " + newAlarm->getTypeString() + " alarm for point " +
